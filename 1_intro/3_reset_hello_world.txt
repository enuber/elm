module Main exposing (main)

import Browser
import Html exposing (Html, div, text)

-- MODEL

type alias Model =
    { phrase : String }

initialModel : Model
initialModel =
    { phrase = "Hello World" }

-- VIEW

view : Model -> Html msg
view model =
    div []
        [ div [] [ text model.phrase ] ]

-- MAIN

main : Program () Model msg
main =
    Browser.sandbox
        { init = initialModel
        , update = \_ model -> model  -- no-op update since it's a static display
        , view = view
        }
